                    </div>

            """, unsafe_allow_html=True)
    else:

            if st.form_submit_button("üíæ Guardar Evento"):

                    st.error("‚ùå El campo 'Enfermedad o Diagn√≥stico' es obligatorio.")

                    st.success("‚úÖ ¬°Evento guardado!")
                    st.rerun()
